Solicite um novo link de recuperação.'; document.getElementById('reset-form').style.display = 'none'; } // Password strength checker const passwordInput = document.getElementById('password'); const strengthBar = document.getElementById('strength-bar'); const strengthText = document.getElementById('strength-text'); passwordInput.addEventListener('input', () => { const password = passwordInput.value; let strength = 0; if (password.length >= 8) strength++; if (password.match(/[a-z]/) && password.match(/[A-Z]/)) strength++; if (password.match(/[0-9]/)) strength++; if (password.match(/[^a-zA-Z0-9]/)) strength++; strengthBar.className = 'password-strength-bar'; if (strength <= 1) { strengthBar.classList.add('strength-weak'); strengthText.textContent = 'Fraca'; strengthText.className = 'text-xs text-red-400 mt-1'; } else if (strength <= 3) { strengthBar.classList.add('strength-medium'); strengthText.textContent = 'Média'; strengthText.className = 'text-xs text-yellow-400 mt-1'; } else { strengthBar.classList.add('strength-strong'); strengthText.textContent = 'Forte'; strengthText.className = 'text-xs text-green-400 mt-1'; } }); // Form submission document.getElementById('reset-form').addEventListener('submit', async (e) => { e.preventDefault(); const password = document.getElementById('password').value; const confirmPassword = document.getElementById('confirm-password').value; const errorMsg = document.getElementById('error-msg'); const btn = e.target.querySelector('button'); const originalText = btn.textContent; // Clear previous errors errorMsg.classList.add('hidden'); // Validate passwords match if (password !== confirmPassword) { errorMsg.classList.remove('hidden'); errorMsg.querySelector('p').textContent = 'As senhas não coincidem.'; return; } // Validate minimum length if (password.length < 8) { errorMsg.classList.remove('hidden'); errorMsg.querySelector('p').textContent = 'A senha deve ter no mínimo 8 caracteres.'; return; } try { btn.disabled = true; btn.textContent = 'Redefinindo...'; const response = await fetch('/api/auth/reset-password', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ token, password }) }); const data = await response.json(); if (response.ok) { // Show success document.getElementById('reset-form').classList.add('hidden'); document.getElementById('success-msg').classList.remove('hidden'); } else { errorMsg.classList.remove('hidden'); errorMsg.querySelector('p').textContent = data.error || 'Erro ao redefinir senha.'; } } catch (error) { console.error('Error:', error); errorMsg.classList.remove('hidden'); errorMsg.querySelector('p').textContent = 'Erro ao redefinir senha. Tente novamente.'; } finally { btn.disabled = false; btn.textContent = originalText; } });
